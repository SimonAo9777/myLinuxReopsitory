#Addition function
add()
{    
	((result=$1+$2))
	echo " The sum of $1 plus $2 equals $result"

}

#substract function
substract()
{
        ((result=$1-$2))
        echo "The sum of $1 substract $2 equals $result"
}

#display menu1

{

}

menu3()

#Beginning of if Statement
if [ $# != 3 ]
